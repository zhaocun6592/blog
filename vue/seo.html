    # 1. 用vue脚手架创建前端框架

     vue create web

    # 2. 引入 normalize.css,用于全局格式化样式

    2.1 引入 normalize.css,用于全局格式化样式

    npm  install --save normalize.css
    1. 2 在main.js中引入
    import 'normalize.css/normalize.css'

    # 3. 配置预渲染seo 插件 详见 客户端预渲染

    # 4. 配置样式less 和全局的样式引入

    # 4.1 引入 less loader 和less 在vue中可以使用less了

    npm i less-loader less --save-dev

    # 4.2 如果需要全局的less 需要引入

    npm i style-resources-loader --save-dev

    然后在vue.config.js中配置less

    const path = require('path')
function resolve (dir) {
  return path.join(__dirname, dir)
}
module.exports = {
  pluginOptions: { // 第三方插件配置
    'style-resources-loader': {
      preProcessor: 'less',
      patterns: [path.resolve(__dirname, './src/assets/styles/vars.less')]
    }
  }
}
