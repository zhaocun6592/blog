# 2. 创建store文件夹

# 里面文件有actionType.js createAction.js reducer.js index.js
# 2.1 actionType 负责创建固定变量
export const CHANGE_NAME = "CHANGE_NAME"
export const CHANGE_COUNT = "CHANGE_COUNT"
# 2.2 createAction 创建action动作的对象
import * as actionTypes from "./actionType"
export const createNameAction = (data)=>({type:actionTypes.CHANGE_NAME,data})
export const createCountAction = (data)=>({type:actionTypes.CHANGE_COUNT,data})
# 2.3 reducer 负责连接react ui和 数据层
import * as actionTypes from "./actionType"  // 引入变量
const initState= {     // 初始化的数据
    count:0,
    age:18,
    name:'zhaoyancun'
}

export default function reducer(state = initState,action){  // reducer是个函数 返回state数据
    console.log(action,actionTypes);
    const {type,data} = action
    switch (type) {
        case actionTypes.CHANGE_NAME:
            return {...state,name:data}
        case actionTypes.CHANGE_COUNT:
            return {...state,count:data}
        default:
            break;
    }
    return state
}
# 2.4 index.js 负责导出整个store数据
import reducer from "./reducer"; // 导入reducer数据
import { createStore } from "redux"; // 创建store仓库
const store = createStore(
  reducer,
  window.__REDUX_DEVTOOLS_EXTENSION__ && window.__REDUX_DEVTOOLS_EXTENSION__()
);
export default store;

# 4. 在组件中使用

import React, { Component } from "react";
import { createNameAction, createCountAction } from "../store/createAction"; //创建action的函数 返回动作的对象
import { connect } from "react-redux"; //负责连接 组件和 store

connect是真正的重点，它是一个科里化函数，意思是先接受两个参数（数据绑定mapStateToProps和事件绑定mapDispatchToProps），再接受一个参数（将要绑定的组件本身）：
mapStateToProps：构建好Redux系统的时候，它会被自动初始化，但是你的React组件并不知道它的存在，因此你需要分拣出你需要的Redux状态，所以你需要绑定一个函数，它的参数是state，简单返回你关心的几个值。mapDispatchToProps：声明好的action作为回调，也可以被注入到组件里，就是通过这个函数，它的参数是dispatch，通过redux的辅助方法bindActionCreator绑定所有action以及参数的dispatch，就可以作为属性在组件里面作为函数简单使用了，不需要手动dispatch。这个mapDispatchToProps是可选的，如果不传这个参数redux会简单把dispatch作为属性注入给组件，可以手动当做store.dispatch使用。这也是为什么要科里化的原因。

//下边的等价于上边
const mapStateToProps = (state)=>{
  return {
    count: state.count, name: state.name
  }
}
const mapDispatchToProps = (dispatch)=>{
  return{
    changeName: data => dispatch(createNameAction(data)),
    changeCount: data => dispatch(createCountAction(data)),
  }

}


export default connect(
  state => ({ count: state.count, name: state.name }),
  dispatch => ({
    changeName: data => dispatch(createNameAction(data)),
    changeCount: data => dispatch(createCountAction(data)),
  })
)(
  class Cun extends Component {
    state = {
      inner: "组件自身的值",
    };
    handelClick = () => {
      this.props.changeName("使用react-redux改变后的名字");
      let {count}  = this.props
      this.props.changeCount(count+1) //使用props 触发方法，已经取值到了
    };
    componentDidMount() {}
    render() {
      const { count, name } = this.props; //使用props传入 方法和store的数据
      return (
        <div>
          <button onClick={this.handelClick}>点击触发dispatch</button>
          <h3>{this.state.inner}</h3>
          {count}
          <h3>{name}</h3>
        </div>
      );
    }
  }
);
